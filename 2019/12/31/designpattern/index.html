// 链式调用

const A = function(value) {
this.value = value
}

A.prototype.add = function(val) {
this.value += val
return this
}

A.prototype.sub = function(val) {
this.value -= val
return this
}

A.prototype.multiply = function(val) {
this.value *= val
return this
}

A.prototype.divide = function(val) {
this.value /= val
return this
}

var calculator = new A(1)
// (1 + 2 - 3) * 5
calculator.add(2).sub(3).multiply(5)
